import { jsPDF } from 'jspdf';
import type { Blueprint, Coordinate } from '@/types/database';

export async function generatePDF(
  blueprint: Blueprint,
  coordinates: Coordinate[]
): Promise {
  const doc = new jsPDF({
    orientation: 'portrait',
    unit: 'mm',
    format: 'a4',
  });

  // Title
  doc.setFontSize(24);
  doc.text(blueprint.title, 105, 20, { align: 'center' });

  // Metadata
  doc.setFontSize(10);
  doc.text(`Generated: ${new Date().toLocaleDateString()}`, 20, 35);
  doc.text(`Diameter: ${blueprint.actual_diameter_cm}cm`, 20, 40);
  doc.text(`Coordinates: ${coordinates.length} points`, 20, 45);

  // Draw blueprint visualization
  const centerX = 105;
  const centerY = 100;
  const radius = 40;

  doc.setDrawColor(74, 103, 65);
  doc.setLineWidth(0.5);
  doc.circle(centerX, centerY, radius);
  doc.circle(centerX, centerY, radius * 0.6, 'S');

  // Plot coordinate points
  coordinates.forEach((coord) => {
    const radians = (coord.angle_degrees * Math.PI) / 180;
    const x = centerX + radius * Math.cos(radians);
    const y = centerY + radius * Math.sin(radians);
    doc.circle(x, y, 1, 'F');
  });

  // Coordinate table
  doc.setFontSize(12);
  doc.text('Coordinate Table', 20, 160);
  
  doc.setFontSize(8);
  let yPos = 170;
  doc.text('Index', 20, yPos);
  doc.text('Angle', 40, yPos);
  doc.text('Pixel X', 60, yPos);
  doc.text('Pixel Y', 80, yPos);
  doc.text('Real X (in)', 100, yPos);
  doc.text('Real Y (in)', 130, yPos);
  doc.text('Position', 160, yPos);

  coordinates.forEach((coord, idx) => {
    yPos += 5;
    if (yPos > 280) {
      doc.addPage();
      yPos = 20;
    }
    doc.text(coord.point_index.toString(), 20, yPos);
    doc.text(coord.angle_degrees.toFixed(1), 40, yPos);
    doc.text(coord.x_coordinate.toFixed(2), 60, yPos);
    doc.text(coord.y_coordinate.toFixed(2), 80, yPos);
    doc.text(coord.real_x_inches?.toFixed(2) || '-', 100, yPos);
    doc.text(coord.real_y_inches?.toFixed(2) || '-', 130, yPos);
    doc.text(coord.position_name || '-', 160, yPos);
  });

  // Footer
  const pageCount = doc.getNumberOfPages();
  for (let i = 1; i <= pageCount; i++) {
    doc.setPage(i);
    doc.setFontSize(8);
    doc.text(
      `WreathBlueprint.com - Page ${i} of ${pageCount}`,
      105,
      290,
      { align: 'center' }
    );
  }

  return doc.output('blob');
}

src/app/api/stripe/checkout/route.ts 
import { NextRequest, NextResponse } from 'next/server';
import Stripe from 'stripe';
import { getUserTier } from '@/lib/tier/validation';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2023-10-16',
});

const TIER_PRICE_MAP = {
  craft: process.env.STRIPE_PRICE_CRAFT_MONTHLY!,
  studio: process.env.STRIPE_PRICE_STUDIO_MONTHLY!,
  atelier: process.env.STRIPE_PRICE_ATELIER_MONTHLY!,
};

export async function POST(request: NextRequest) {
  try {
    const { user, profile } = await getUserTier();
    const { tier } = await request.json();
    
    if (!TIER_PRICE_MAP[tier]) {
      return NextResponse.json(
        { error: 'Invalid tier' },
        { status: 400 }
      );
    }
    
    // Create or retrieve Stripe customer
    let customerId = profile.stripe_customer_id;
    
    if (!customerId) {
      const customer = await stripe.customers.create({
        email: user.email,
        metadata: {
          supabase_id: user.id,
        },
      });
      customerId = customer.id;
      
      // Update profile with customer ID
      const supabase = createClient();
      await supabase
        .from('profiles')
        .update({ stripe_customer_id: customerId })
        .eq('id', user.id);
    }
    
    // Create checkout session
    const session = await stripe.checkout.sessions.create({
      customer: customerId,
      line_items: [
        {
          price: TIER_PRICE_MAP[tier],
          quantity: 1,
        },
      ],
      mode: 'subscription',
      success_url: `${process.env.NEXT_PUBLIC_APP_URL}/dashboard?upgrade=success`,
      cancel_url: `${process.env.NEXT_PUBLIC_APP_URL}/pricing?upgrade=cancelled`,
      metadata: {
        user_id: user.id,
        tier,
      },
    });
    
    return NextResponse.json({ url: session.url });
  } catch (error) {
    return NextResponse.json(
      { error: error.message },
      { status: 500 }
    );
  }
}
src/app/api/stripe/webhook/route.ts 
import { NextRequest, NextResponse } from 'next/server';
import Stripe from 'stripe';
import { createServiceClient } from '@/lib/supabase/server';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2023-10-16',
});

export async function POST(request: NextRequest) {
  const body = await request.text();
  const sig = request.headers.get('stripe-signature')!;
  
  let event: Stripe.Event;
  
  try {
    event = stripe.webhooks.constructEvent(
      body,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET!
    );
  } catch (err) {
    return NextResponse.json(
      { error: 'Webhook signature verification failed' },
      { status: 400 }
    );
  }
  
  const supabase = createServiceClient();
  
  switch (event.type) {
    case 'checkout.session.completed': {
      const session = event.data.object as Stripe.Checkout.Session;
      const userId = session.metadata?.user_id;
      const tier = session.metadata?.tier;
      
      if (userId && tier) {
        // Update user tier
        await supabase
          .from('profiles')
          .update({ subscription_tier: tier })
          .eq('id', userId);
          
        // Create subscription record
        await supabase.from('subscriptions').insert({
          user_id: userId,
          stripe_subscription_id: session.subscription,
          tier,
          status: 'active',
        });
      }
      break;
    }
    
    case 'customer.subscription.deleted': {
      const subscription = event.data.object as Stripe.Subscription;
      
      // Downgrade to free tier
      await supabase
        .from('subscriptions')
        .update({ status: 'cancelled' })
        .eq('stripe_subscription_id', subscription.id);
        
      const { data } = await supabase
        .from('subscriptions')
        .select('user_id')
        .eq('stripe_subscription_id', subscription.id)
        .single();
        
      if (data) {
        await supabase
          .from('profiles')
          .update({ subscription_tier: 'bloom' })
          .eq('id', data.user_id);
      }
      break;
    }
  }
  
  return NextResponse.json({ received: true });
}
