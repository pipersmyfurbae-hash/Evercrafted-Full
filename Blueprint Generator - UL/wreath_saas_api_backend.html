src/types/database.ts 
// Generated from Supabase schema
// Run: supabase gen types typescript --local > src/types/database.ts

export type SubscriptionTier = 'bloom' | 'craft' | 'studio' | 'atelier';
export type ProcessingStatus = 'pending' | 'processing' | 'completed' | 'failed';
export type ExportFormat = 'pdf' | 'svg' | 'json' | 'csv';

export interface Profile {
  id: string;
  email: string;
  full_name: string | null;
  subscription_tier: SubscriptionTier;
  stripe_customer_id: string | null;
  credits_remaining: number;
  credits_reset_at: string;
  created_at: string;
  updated_at: string;
}

export interface Blueprint {
  id: string;
  user_id: string;
  title: string;
  description: string | null;
  image_url: string | null;
  image_width: number | null;
  image_height: number | null;
  actual_diameter_cm: number | null;
  detected_diameter_px: number | null;
  center_x: number;
  center_y: number;
  outer_radius: number;
  inner_radius: number;
  processing_status: ProcessingStatus;
  confidence_score: number | null;
  coordinate_count: number;
  metadata: Record<string, any> | null;
  created_at: string;
  updated_at: string;
}

export interface Coordinate {
  id: string;
  blueprint_id: string;
  point_index: number;
  angle_degrees: number;
  x_coordinate: number;
  y_coordinate: number;
  real_x_inches: number | null;
  real_y_inches: number | null;
  position_name: string | null;
  assigned_element_id: string | null;
  created_at: string;
}

export interface TierLimits {
  blueprints_per_month: number;
  export_formats: ExportFormat[];
  api_access: boolean;
  batch_processing: boolean;
  coordinate_precision: number[];
}
