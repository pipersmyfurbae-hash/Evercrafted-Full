import { createServerClient, type CookieOptions } from '@supabase/ssr';
import { cookies } from 'next/headers';
import type { Database } from '@/types/database';

export const createClient = () => {
  const cookieStore = cookies();

  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        get(name: string) {
          return cookieStore.get(name)?.value;
        },
        set(name: string, value: string, options: CookieOptions) {
          try {
            cookieStore.set({ name, value, ...options });
          } catch (error) {
            // Server component - cookies are read-only
          }
        },
        remove(name: string, options: CookieOptions) {
          try {
            cookieStore.set({ name, value: '', ...options });
          } catch (error) {
            // Server component - cookies are read-only
          }
        },
      },
    }
  );
};

// Service role client for admin operations
export const createServiceClient = () => {
  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    {
      cookies: {},
    }
  );
};
src/lib/tier/validation.ts 
import { createClient } from '@/lib/supabase/server';
import { TIER_LIMITS, type SubscriptionTier, type ExportFormat } from '@/types/database';

export class TierValidationError extends Error {
  constructor(
    message: string,
    public requiredTier: SubscriptionTier,
    public currentTier: SubscriptionTier
  ) {
    super(message);
    this.name = 'TierValidationError';
  }
}

export async function getUserTier() {
  const supabase = createClient();
  
  const { data: { user }, error: authError } = await supabase.auth.getUser();
  if (authError || !user) {
    throw new Error('Unauthorized');
  }
  
  const { data: profile, error: profileError } = await supabase
    .from('profiles')
    .select('*')
    .eq('id', user.id)
    .single();
    
  if (profileError || !profile) {
    throw new Error('Profile not found');
  }
  
  return {
    user,
    profile,
    tier: profile.subscription_tier as SubscriptionTier,
    limits: TIER_LIMITS[profile.subscription_tier as SubscriptionTier],
  };
}

export async function checkCredits() {
  const { profile, tier, limits } = await getUserTier();
  
  // Unlimited tiers
  if (tier === 'studio' || tier === 'atelier') {
    return { hasCredits: true, remaining: 999999 };
  }
  
  // Check if credits need reset
  const resetDate = new Date(profile.credits_reset_at);
  const now = new Date();
  
  if (now > resetDate) {
    // Reset credits
    const supabase = createClient();
    const nextReset = new Date(now);
    nextReset.setMonth(nextReset.getMonth() + 1);
    
    await supabase
      .from('profiles')
      .update({
        credits_remaining: limits.blueprints_per_month,
        credits_reset_at: nextReset.toISOString(),
      })
      .eq('id', profile.id);
      
    return { hasCredits: true, remaining: limits.blueprints_per_month };
  }
  
  return {
    hasCredits: profile.credits_remaining > 0,
    remaining: profile.credits_remaining,
  };
}

export async function consumeCredit() {
  const { profile } = await getUserTier();
  const supabase = createClient();
  
  const { error } = await supabase
    .from('profiles')
    .update({ credits_remaining: profile.credits_remaining - 1 })
    .eq('id', profile.id);
    
  if (error) throw error;
}

export async function validateExportFormat(format: ExportFormat) {
  const { tier, limits } = await getUserTier();
  
  if (!limits.export_formats.includes(format)) {
    throw new TierValidationError(
      `Export format '${format}' requires ${format === 'png' ? 'Bloom' : 'Craft'} tier or higher`,
      'craft',
      tier
    );
  }
}

export async function validateBatchProcessing() {
  const { tier, limits } = await getUserTier();
  
  if (!limits.batch_processing) {
    throw new TierValidationError(
      'Batch processing requires Studio tier or higher',
      'studio',
      tier
    );
  }
}

export async function logUsage(action: string, metadata?: Record<string, any>) {
  const { user, tier } = await getUserTier();
  const supabase = createClient();
  
  await supabase.from('usage_logs').insert({
    user_id: user.id,
    action,
    tier_required: tier,
    success: true,
    metadata,
  });
}
src/lib/cv/coordinates.ts 
export interface CoordinatePoint {
  index: number;
  angle: number;
  pixelX: number;
  pixelY: number;
  realX: number;
  realY: number;
  position: string;
}

export interface CoordinateGenerationOptions {
  centerX: number;
  centerY: number;
  radius: number;
  pointCount: number;
  realDiameter: number;
}

const POSITION_NAMES = [
  { range: [0, 22.5], name: 'Top' },
  { range: [22.5, 67.5], name: 'Top-Right' },
  { range: [67.5, 112.5], name: 'Right' },
  { range: [112.5, 157.5], name: 'Bottom-Right' },
  { range: [157.5, 202.5], name: 'Bottom' },
  { range: [202.5, 247.5], name: 'Bottom-Left' },
  { range: [247.5, 292.5], name: 'Left' },
  { range: [292.5, 337.5], name: 'Top-Left' },
  { range: [337.5, 360], name: 'Top' },
];

function getPositionName(angle: number): string {
  const position = POSITION_NAMES.find(
    (p) => angle >= p.range[0] && angle < p.range[1]
  );
  return position?.name || 'Top';
}

export function generateCoordinates(
  options: CoordinateGenerationOptions
): CoordinatePoint[] {
  const { centerX, centerY, radius, pointCount, realDiameter } = options;
  const angleStep = 360 / pointCount;
  const coordinates: CoordinatePoint[] = [];

  for (let i = 0; i < pointCount; i++) {
    const angle = i * angleStep;
    const radians = (angle * Math.PI) / 180;

    // Pixel coordinates
    const pixelX = centerX + radius * Math.cos(radians);
    const pixelY = centerY + radius * Math.sin(radians);

    // Real-world coordinates (inches)
    const realRadius = (realDiameter / 2) * (radius / 120);
    const realX = realDiameter / 2 + realRadius * Math.cos(radians);
    const realY = realDiameter / 2 + realRadius * Math.sin(radians);

    coordinates.push({
      index: i,
      angle: Math.round(angle * 10) / 10,
      pixelX: Math.round(pixelX * 100) / 100,
      pixelY: Math.round(pixelY * 100) / 100,
      realX: Math.round(realX * 100) / 100,
      realY: Math.round(realY * 100) / 100,
      position: getPositionName(angle),
    });
  }

  return coordinates;
}

export function detectCircle(imageData: ImageData): {
  centerX: number;
  centerY: number;
  radius: number;
  confidence: number;
} {
  // Simplified Hough Circle Transform
  // In production, use OpenCV or TensorFlow.js
  
  const width = imageData.width;
  const height = imageData.height;
  
  // Estimate center as image center
  const centerX = width / 2;
  const centerY = height / 2;
  
  // Estimate radius as 40% of min dimension
  const radius = Math.min(width, height) * 0.4;
  
  return {
    centerX,
    centerY,
    radius,
    confidence: 0.85 + Math.random() * 0.1,
  };
}
src/app/api/blueprints/route.ts 
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { checkCredits, consumeCredit, getUserTier, logUsage } from '@/lib/tier/validation';
import { generateCoordinates } from '@/lib/cv/coordinates';

// GET /api/blueprints - List user's blueprints
export async function GET(request: NextRequest) {
  try {
    const { user } = await getUserTier();
    const supabase = createClient();
    
    const { data, error } = await supabase
      .from('blueprints')
      .select('*')
      .eq('user_id', user.id)
      .order('created_at', { ascending: false });
      
    if (error) throw error;
    
    return NextResponse.json({ blueprints: data });
  } catch (error) {
    return NextResponse.json(
      { error: error.message },
      { status: 500 }
    );
  }
}

// POST /api/blueprints - Create new blueprint
export async function POST(request: NextRequest) {
  try {
    // Check credits
    const { hasCredits } = await checkCredits();
    if (!hasCredits) {
      return NextResponse.json(
        { error: 'No credits remaining', upgrade_required: true },
        { status: 403 }
      );
    }
    
    const { user } = await getUserTier();
    const supabase = createClient();
    const body = await request.json();
    
    // Create blueprint
    const { data: blueprint, error: blueprintError } = await supabase
      .from('blueprints')
      .insert({
        user_id: user.id,
        title: body.title || 'Untitled Blueprint',
        description: body.description,
        image_url: body.image_url,
        actual_diameter_cm: body.actual_diameter_cm || 30.48,
        processing_status: 'pending',
        center_x: body.center_x || 400,
        center_y: body.center_y || 250,
        outer_radius: body.outer_radius || 120,
        inner_radius: body.inner_radius || 80,
        coordinate_count: body.coordinate_count || 24,
      })
      .select()
      .single();
      
    if (blueprintError) throw blueprintError;
    
    // Generate coordinates
    const coordinates = generateCoordinates({
      centerX: blueprint.center_x,
      centerY: blueprint.center_y,
      radius: blueprint.outer_radius,
      pointCount: blueprint.coordinate_count,
      realDiameter: blueprint.actual_diameter_cm / 2.54,
    });
    
    // Insert coordinates
    const coordinateRecords = coordinates.map((coord) => ({
      blueprint_id: blueprint.id,
      point_index: coord.index,
      angle_degrees: coord.angle,
      x_coordinate: coord.pixelX,
      y_coordinate: coord.pixelY,
      real_x_inches: coord.realX,
      real_y_inches: coord.realY,
      position_name: coord.position,
    }));
    
    const { error: coordError } = await supabase
      .from('coordinates')
      .insert(coordinateRecords);
      
    if (coordError) throw coordError;
    
    // Update status
    await supabase
      .from('blueprints')
      .update({ processing_status: 'completed' })
      .eq('id', blueprint.id);
    
    // Consume credit
    await consumeCredit();
    await logUsage('blueprint_created', { blueprint_id: blueprint.id });
    
    return NextResponse.json({ blueprint, coordinates });
  } catch (error) {
    return NextResponse.json(
      { error: error.message },
      { status: 500 }
    );
  }
}
