Coordinate System Reference
/* QACS Polar Coordinate Spec · EC_CANON_V1 */ Origin: Center of wreath base Angle 0°: 12:00 position (top) Direction: Clockwise (CW) Radius: Normalized 0–1 (1 = wreath edge) Canonical: { rNorm, theta } — stored as radius_norm + angle_deg Arc Bands: inner: 0.00–0.33 midpoint 0.16 // base, anchor mid: 0.33–0.66 midpoint 0.50 // focal, body outer: 0.66–1.00 midpoint 0.83 // accent, drape Clock Mapping: 12:00 → 0° 3:00 → 90° 6:00 → 180° 9:00 → 270° Quadrant Mapping: NE: 0°–90° SE: 90°–180° SW: 180°–270° NW: 270°–360° XY Canvas Conversion: x = cx + r * sin(angle_rad) * radius_norm y = cy - r * cos(angle_rad) * radius_norm
TypeScript Interface · WreathPlacement
interface WreathPlacement { id: string; element: string; category: 'focal'|'greenery'|'filler'|'accent'|'base'|'ribbon'; angle_deg: number; // 0–359, 0 = 12:00 CW clock: string; // "3:00" radius: 'inner'|'mid'|'outer'; radius_norm: number; // 0.16 | 0.50 | 0.83 density: 'low'|'medium'|'high'; stem_count: number; quadrant: 'NE'|'SE'|'SW'|'NW'; wreath_size_in: number; xy: { x: number; y: number }; } interface QACSBlueprint { wreath_id: string; design_name: string; size_in: number; element_count: number; symmetry_mode: string; notes: string; placements: WreathPlacement[]; }
